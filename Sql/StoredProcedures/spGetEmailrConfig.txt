IF EXISTS (SELECT * FROM sys.objects WHERE type = 'P' AND OBJECT_ID = OBJECT_ID('dbo.spGetEmailConfig'))
   drop proc [dbo].[spGetEmailConfig]
GO
CREATE PROCEDURE [dbo].[spGetEmailConfig] 
AS
BEGIN
  select cfg.[Enabled]
         ,cfg.[ClientId]
		 ,cfg.[EncryptedClientSecret]
	     ,cfg.[TenantId]
    from [dbo].[EmailConfig] cfg
   where cfg.[Enabled] = 1
END
